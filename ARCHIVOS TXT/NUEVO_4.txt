    <script>
        // Menú móvil
        const hamburger = document.querySelector('.hamburger');
        const mobileMenu = document.querySelector('.mobile-menu');
        const closeMenu = document.querySelector('.close-menu');
        hamburger.addEventListener('click', () => {
            mobileMenu.classList.toggle('active');
        });
        closeMenu.addEventListener('click', () => {
            mobileMenu.classList.remove('active');
        });

        // Animaciones
        document.querySelectorAll('.pop-animation').forEach((el, index) => {
            setTimeout(() => {
                el.classList.add('animate');
            }, index * 200);
        });

        // Acordeón
        document.querySelectorAll('.accordion-title').forEach(item => {
            item.addEventListener('click', () => {
                const content = item.nextElementSibling;
                content.classList.toggle('active');
                item.classList.toggle('active');
            });
        });

        // Navegación descendente
        const sections = ['inicio', 'introduccion', 'requisitos', 'upci', 'como_inicio', 'es_complicado', 'documentos'];
        const navLinks = document.getElementById('nav-links');
        const mobileNavLinks = document.getElementById('mobile-nav-links');
        window.addEventListener('scroll', () => {
            let currentSection = '';
            sections.forEach(section => {
                const sectionElement = document.getElementById(section);
                if (sectionElement.getBoundingClientRect().top <= 100) {
                    currentSection = section;
                }
            });
            const currentIndex = sections.indexOf(currentSection);
            navLinks.innerHTML = '';
            mobileNavLinks.innerHTML = '';
            sections.slice(currentIndex + 1).forEach(section => {
                const link = document.createElement('a');
                link.href = `#${section}`;
                link.className = 'nav-item';
                link.textContent = section.charAt(0).toUpperCase() + section.slice(1).replace('_', ' ');
                navLinks.appendChild(link);
                const mobileLink = document.createElement('a');
                mobileLink.href = `#${section}`;
                mobileLink.className = 'mobile-nav-item';
                mobileLink.textContent = section.charAt(0).toUpperCase() + section.slice(1).replace('_', ' ');
                mobileNavLinks.appendChild(mobileLink);
            });
        });
    </script>
